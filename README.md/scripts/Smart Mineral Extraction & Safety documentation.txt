-- Function to check restrictions
CREATE OR REPLACE FUNCTION fn_check_restriction RETURN VARCHAR2 IS
    v_day VARCHAR2(10);
    v_today DATE := TRUNC(SYSDATE);
    v_is_holiday NUMBER;
BEGIN
    SELECT TO_CHAR(v_today, 'DY', 'NLS_DATE_LANGUAGE=ENGLISH') INTO v_day FROM DUAL;
    SELECT COUNT(*) INTO v_is_holiday FROM holiday WHERE holiday_date = v_today;

    IF v_is_holiday > 0 THEN
        RETURN 'BLOCK_HOLIDAY';
    ELSIF v_day IN ('MON','TUE','WED','THU','FRI') THEN
        RETURN 'BLOCK_WEEKDAY';
    ELSE
        RETURN 'ALLOW';
    END IF;
EXCEPTION
    WHEN OTHERS THEN
        RETURN 'ALLOW';
END fn_check_restriction;
/

-- Procedure to load dummy extractions
CREATE OR REPLACE PROCEDURE load_dummy_extractions(p_num IN NUMBER) IS
    TYPE t_site_id IS TABLE OF extraction.site_id%TYPE INDEX BY PLS_INTEGER;
    TYPE t_mineral_id IS TABLE OF extraction.mineral_id%TYPE INDEX BY PLS_INTEGER;
    TYPE t_date_extracted IS TABLE OF extraction.date_extracted%TYPE INDEX BY PLS_INTEGER;
    TYPE t_qty IS TABLE OF extraction.quantity_tons%TYPE INDEX BY PLS_INTEGER;
    TYPE t_shift IS TABLE OF extraction.shift%TYPE INDEX BY PLS_INTEGER;
    TYPE t_operator IS TABLE OF extraction.operator_id%TYPE INDEX BY PLS_INTEGER;

    v_site   t_site_id;
    v_min    t_mineral_id;
    v_date   t_date_extracted;
    v_qtys   t_qty;
    v_shift  t_shift;
    v_ops    t_operator;
BEGIN
    FOR i IN 1..p_num LOOP
        v_site(i) := MOD(i,10)+1;
        v_min(i) := MOD(i,6)+1;
        v_date(i) := TRUNC(SYSDATE) - MOD(i,365);
        v_qtys(i) := ROUND(DBMS_RANDOM.VALUE(0.1,100),3);
        v_shift(i) := CASE WHEN MOD(i,2)=0 THEN 'Day' ELSE 'Night' END;
        v_ops(i) := MOD(i,50)+1; -- Changed from 250 to 50 to match worker count
    END LOOP;

    FORALL idx IN INDICES OF v_site
        INSERT INTO extraction(site_id, mineral_id, date_extracted, quantity_tons, shift, operator_id)
        VALUES (v_site(idx), v_min(idx), v_date(idx), v_qtys(idx), v_shift(idx), v_ops(idx));

    COMMIT;
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE;
END load_dummy_extractions;
/

-- Procedure to promote technicians
CREATE OR REPLACE PROCEDURE promote_technicians(p_years IN NUMBER, p_updated OUT NUMBER) IS
    CURSOR c_workers IS
        SELECT worker_id, name, role, hire_date FROM worker WHERE role = 'Technician' FOR UPDATE;
    TYPE t_worker_tab IS TABLE OF c_workers%ROWTYPE;
    v_workers t_worker_tab;
BEGIN
    OPEN c_workers;
    FETCH c_workers BULK COLLECT INTO v_workers LIMIT 100;
    CLOSE c_workers;

    p_updated := 0;
    FOR i IN 1..v_workers.COUNT LOOP
        IF v_workers(i).hire_date <= ADD_MONTHS(SYSDATE, -12*p_years) THEN
            UPDATE worker SET role = 'Senior Technician' WHERE worker_id = v_workers(i).worker_id;
            p_updated := p_updated + 1;
        END IF;
    END LOOP;
    COMMIT;
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE;
END promote_technicians;
/

-- Procedure to calculate revenue
CREATE OR REPLACE PROCEDURE calculate_revenue (
    p_extraction_id IN NUMBER
) IS
BEGIN
    UPDATE extraction e
    SET e.revenue_calculated = e.quantity_tons * (
        SELECT m.market_price
        FROM mineral m
        WHERE m.mineral_id = e.mineral_id
    )
    WHERE e.extraction_id = p_extraction_id;
    COMMIT;
END;

-- Mine Site trigger
CREATE OR REPLACE TRIGGER trg_mine_site_audit_restrict
FOR INSERT OR UPDATE OR DELETE ON mine_site
COMPOUND TRIGGER

  v_user   VARCHAR2(100) := NVL(SYS_CONTEXT('USERENV','SESSION_USER'),'UNKNOWN');
  v_status VARCHAR2(30);

  BEFORE EACH ROW IS
  BEGIN
    v_status := fn_check_restriction();
    IF v_status LIKE 'BLOCK%' THEN
      audit_pkg.log_action(v_user, 'BLOCKED_DML', 'MINE_SITE',
                           NVL(TO_CHAR(:NEW.site_id), TO_CHAR(:OLD.site_id)), 'N',
                           'Restricted: '||v_status);
      RAISE_APPLICATION_ERROR(-20010, 'DML on MINE_SITE not allowed: '||v_status);
    END IF;
  END BEFORE EACH ROW;

  AFTER EACH ROW IS
  BEGIN
    IF INSERTING THEN
      audit_pkg.log_action(v_user, 'INSERT', 'MINE_SITE', TO_CHAR(:NEW.site_id), 'Y', NULL);
    ELSIF UPDATING THEN
      audit_pkg.log_action(v_user, 'UPDATE', 'MINE_SITE', TO_CHAR(:NEW.site_id), 'Y', NULL);
    ELSIF DELETING THEN
      audit_pkg.log_action(v_user, 'DELETE', 'MINE_SITE', TO_CHAR(:OLD.site_id), 'Y', NULL);
    END IF;
  END AFTER EACH ROW;

END trg_mine_site_audit_restrict;
/

-- Mineral trigger
CREATE OR REPLACE TRIGGER trg_mineral_audit
  AFTER INSERT OR UPDATE OR DELETE ON mineral
  FOR EACH ROW
DECLARE
  v_user VARCHAR2(100) := NVL(SYS_CONTEXT('USERENV','SESSION_USER'),'UNKNOWN');
BEGIN
  IF INSERTING THEN
    audit_pkg.log_action(v_user,'INSERT','MINERAL',TO_CHAR(:NEW.mineral_id),'Y', 'New mineral: '||:NEW.mineral_name);

                         'Price changed from '||:OLD.market_price||' to '||:NEW.market_price);
  ELSIF DELETING THEN
    audit_pkg.log_action(v_user,'DELETE','MINERAL',TO_CHAR(:OLD.mineral_id),'Y', 'Removed: '||:OLD.mineral_name);
  END IF;
END;
/

-- Worker trigger
CREATE OR REPLACE TRIGGER trg_worker_full
FOR INSERT OR UPDATE OR DELETE ON worker
COMPOUND TRIGGER

  v_user VARCHAR2(100) := NVL(SYS_CONTEXT('USERENV','SESSION_USER'),'UNKNOWN');
  v_status VARCHAR2(30);

  BEFORE EACH ROW IS
  BEGIN
    v_status := fn_check_restriction();
    IF v_status LIKE 'BLOCK%' THEN
      audit_pkg.log_action(v_user,'BLOCKED_DML','WORKER',
                           NVL(TO_CHAR(:NEW.worker_id),TO_CHAR(:OLD.worker_id)),'N','Restricted: '||v_status);
      RAISE_APPLICATION_ERROR(-20011, 'Worker DML blocked: '||v_status);
    END IF;

    IF UPDATING AND :OLD.safety_status != :NEW.safety_status AND :NEW.safety_status = 'CLEARED' THEN
      audit_pkg.log_action(v_user,'SAFETY_CLEAR','WORKER',TO_CHAR(:NEW.worker_id),'Y','Safety status reset to CLEARED');
    END IF;
  END BEFORE EACH ROW;

  AFTER EACH ROW IS
  BEGIN
    IF INSERTING  THEN audit_pkg.log_action(v_user,'INSERT','WORKER', TO_CHAR(:NEW.worker_id),'Y',NULL);
    ELSIF UPDATING THEN audit_pkg.log_action(v_user,'UPDATE','WORKER', TO_CHAR(:NEW.worker_id),'Y',NULL);
    ELSIF DELETING THEN audit_pkg.log_action(v_user,'DELETE','WORKER', TO_CHAR(:OLD.worker_id),'Y',NULL);
    END IF;
  END AFTER EACH ROW;

END trg_worker_full;
/

-- Machinery trigger
CREATE OR REPLACE TRIGGER trg_machinery_full
FOR INSERT OR UPDATE OR DELETE ON machinery
COMPOUND TRIGGER

  v_user VARCHAR2(100) := NVL(SYS_CONTEXT('USERENV','SESSION_USER'), 'UNKNOWN');

  BEFORE EACH ROW IS
  BEGIN
    IF fn_check_restriction() LIKE 'BLOCK%' THEN
      RAISE_APPLICATION_ERROR(-20012, 'Machinery DML blocked on weekdays/holidays');
    END IF;
  END BEFORE EACH ROW;

  AFTER EACH ROW IS
  BEGIN
    IF INSERTING OR UPDATING THEN
      IF :NEW.next_maintenance < TRUNC(SYSDATE) THEN
        audit_pkg.log_action(v_user, 'MAINT_OVERDUE', 'MACHINERY', TO_CHAR(:NEW.machine_id), 'Y',
          'Next maintenance overdue since ' || TO_CHAR(:NEW.next_maintenance,'YYYY-MM-DD'));
      END IF;
    END IF;

    IF INSERTING THEN
      audit_pkg.log_action(v_user,'INSERT','MACHINERY', TO_CHAR(:NEW.machine_id),'Y',NULL);
    ELSIF UPDATING THEN
      audit_pkg.log_action(v_user,'UPDATE','MACHINERY', TO_CHAR(:NEW.machine_id),'Y',NULL);
    ELSIF DELETING THEN
      audit_pkg.log_action(v_user,'DELETE','MACHINERY', TO_CHAR(:OLD.machine_id),'Y',NULL);
    END IF;
  END AFTER EACH ROW;

END trg_machinery_full;

-- Extraction trigger for revenue calculation
CREATE OR REPLACE TRIGGER trg_extraction_revenue
  BEFORE INSERT OR UPDATE ON extraction
  FOR EACH ROW
DECLARE
  v_price mineral.market_price%TYPE;
BEGIN
  BEGIN
    SELECT market_price INTO v_price
    FROM mineral WHERE mineral_id = :NEW.mineral_id;
    :NEW.revenue_calculated := ROUND(:NEW.quantity_tons * v_price, 2);
  EXCEPTION
    WHEN NO_DATA_FOUND THEN :NEW.revenue_calculated := NULL;
  END;
  :NEW.quantity_tons := NVL(:NEW.quantity_tons, 0);
END;
/

-- Extraction audit trigger
CREATE OR REPLACE TRIGGER trg_extraction_audit_restrict
FOR INSERT OR UPDATE OR DELETE ON extraction
COMPOUND TRIGGER

  v_user VARCHAR2(100) := NVL(SYS_CONTEXT('USERENV','SESSION_USER'),'UNKNOWN');

  BEFORE EACH ROW IS
  BEGIN
    IF fn_check_restriction() LIKE 'BLOCK%' THEN
      RAISE_APPLICATION_ERROR(-20001, 'Extraction DML blocked (weekday/holiday rule)');
    END IF;
  END BEFORE EACH ROW;

  AFTER EACH ROW IS
  BEGIN
    IF INSERTING  THEN audit_pkg.log_action(v_user,'INSERT','EXTRACTION',TO_CHAR(:NEW.extraction_id),'Y',
                                            'Qty='||:NEW.quantity_tons||' tons, Revenue='||:NEW.revenue_calculated);
    ELSIF UPDATING THEN audit_pkg.log_action(v_user,'UPDATE','EXTRACTION',TO_CHAR(:NEW.extraction_id),'Y',NULL);
    ELSIF DELETING THEN audit_pkg.log_action(v_user,'DELETE','EXTRACTION',TO_CHAR(:OLD.extraction_id),'Y',NULL);
    END IF;
  END AFTER EACH ROW;

END trg_extraction_audit_restrict;
/

-- Safety report trigger
CREATE OR REPLACE TRIGGER trg_safety_report_status
  AFTER INSERT OR UPDATE ON safety_report
  FOR EACH ROW
BEGIN
  IF :NEW.severity_level >= 3 AND :NEW.worker_id IS NOT NULL THEN
    UPDATE worker
       SET safety_status = 'UNDER_REVIEW'
     WHERE worker_id = :NEW.worker_id
       AND (safety_status IS NULL OR safety_status != 'UNDER_REVIEW');

    audit_pkg.log_action(USER,'SAFETY_ESCALATION','WORKER',TO_CHAR(:NEW.worker_id),'Y',
                         'Severity '||:NEW.severity_level||' - status UNDER_REVIEW');
  END IF;
END;
/
